const CrcModel = require("./crc-model");
const CrcContext = require("./crc-context");
const {astNodeFactoryMap} = require("./descriptors");
const {isEqual, uniqBy} = require("lodash");

const getContext = (result, descriptors) => {
  const contexts = [];
  descriptors.forEach((descriptorFactory) => {
    const context = CrcContext.create(
      result,
      descriptorFactory
    );

    contexts.push(context);
  });
  return uniqBy(contexts, isEqual);
};

const getAllContexts = (results, descriptors) => {
  let contexts = [];
  results.forEach((result) => {
    const ctxt = getContext(result, descriptors);

    // Console.log(ctxt);

    contexts = contexts.concat(ctxt);
  });
  return contexts;
};

/**
 * Generates an array of `CrcClasses`.
 */

class CrcReporter {
  constructor () {
    this.crcModels = [];
    this.descriptors = astNodeFactoryMap;
    this.map = new Map();
  }

  /**
   * Create a report from all ECMAScript resources.
   *
   * @param {array.<Result>} results - A list of result objects.
   *
   * @returns {array.<CrcClass>} A list of CrcClasses.
   */

  report (results) {
    const crcClasses = [];
    getAllContexts(results, this.descriptors).forEach((context) => {
      // Console.log(context.descriptor);

      const descriptor = this.descriptors.get(context.descriptor);

      const crcClass = descriptor.factory(context);

      crcClasses.push(crcClass);
    });

    return crcClasses.map((crcClass) => new CrcModel({
      "class": crcClass,
      "collaborators": [],
      "responsibilities": []
    }));
  }
}

module.exports = CrcReporter;
